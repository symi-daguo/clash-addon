#include <tunables/global>

profile proxy-service flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # 基本能力
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # 访问插件内的文件和选项
  /data/** rw,

  # Nginx配置和日志
  /etc/nginx/** rw,
  /var/log/nginx/** rwk,

  # Clash
  /opt/clash/** rwk,

  # 为Clash服务启动新的权限配置
  /usr/bin/proxy-service cx -> proxy-service,
  /opt/clash/clash cx -> clash,

  profile proxy-service flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # 从S6-Overlay接收信号
    signal (receive) peer=*_proxy-service,

    # 访问必要文件
    /data/** rw,
    /opt/clash/** rwk,
    /share/** rw,
    /etc/nginx/** rw,
    /tmp/** rwk,

    # 基本命令
    /usr/bin/proxy-service r,
    /bin/bash rix,
    /bin/sh rix,
    /usr/bin/curl rix,
    /usr/bin/jq rix,
    /usr/bin/tempio rix,
    /usr/bin/cat rix,
    /usr/bin/grep rix,
    /usr/bin/sed rix,
    /usr/bin/base64 rix,
    /usr/bin/rm rix,
    /etc/passwd r,
    /dev/tty rw,
  }

  profile clash flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # 从S6-Overlay接收信号
    signal (receive) peer=*_proxy-service,

    # 网络权限
    network inet stream,
    network inet dgram,
    network inet6 stream,
    network inet6 dgram,

    # 文件访问
    /opt/clash/** rwk,
    /opt/clash/clash mr,
    /opt/clash/config/** rw,
    /dev/tty rw,
  }
}
